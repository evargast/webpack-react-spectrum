import { action } from "@storybook/addon-actions";
import { Canvas, Meta, Story } from "@storybook/addon-docs/blocks";

import App from "./App";

<Meta title="MDX/App" component={App} />

# App

With `MDX` we can define a story for `App` right in the middle of our
Markdown documentation.

This repo also supports Storybook Controls, which allows
you to manipulate the component props straight from your Storybook Canvas! It auto-generates the controls based on the
**args** passed to the Story. You can
further customize this by passing **argTypes** check the [docs
here](https://storybook.js.org/docs/react/essentials/controls)
for more info!

<!--- This is your MDX story template, works just like TSX! -->

export const ComponentStory = props => <App {...props} />;

<Canvas>
    <Story
        name="Enabled"
        args={{
            text: "Hello",
            disabled: false,
            onClick: action("Button clicked!"),
        }}
        argTypes={{ disabled: { control: { type: "boolean" } } }}
    >
        {ComponentStory.bind({})}
    </Story>
    <Story
        name="Disabled"
        args={{
            text: "World",
            disabled: true,
        }}
    >
        {ComponentStory.bind({})}
    </Story>
</Canvas>
